Question: Implement search operations on the array
Solution:
#include <stdio.h>
#include <stdlib.h>
//Implement search operations on the array

void binarySearch(int a[], int *size)
{
    int x;
    int n= *size;
    printf("Enter the element to be searched :\n");
    scanf("%d", &x);
    int first=0,last=n-1,mid=(first+last)/2;
    while(first<=last)
    {
        if(a[mid]<x)
        {
            first=mid+1;
            mid=(first+last)/2;
        }
        else if(a[mid]==x)
        {
            printf("%d found at %dth index of the array.\n", x, mid);
            break;
        }
        else
        {
            last=mid-1;
            mid=(first+last)/2;
        }
        if(first>last)
        {
            printf("%d not found in the array. Not present here.\n", x);
        }
    }
}
void linearSearch(int a[], int *size)
{
    int x,count=0;
    int n= *size;
    printf("Enter the element to be searched in the array:\n");
    scanf("%d", &x);
    for(int i=0;i<n;i++)
    {
        if(a[i]==x)
        {
            printf("%d found at %dth index of the array.\n", x, i);
            count+=1;
            break;
        }
    }
    if(count==0)
    {
    printf("%d not found in the array. Not present here.\n",x);
    }
}
void display(int a[], int *size)    //Function to display the elements of the array
{
    int n= *size;
    printf("The elements of the Array are:\n");
    for(int i=0; i<n;i++)
    {
        printf("%d\n",a[i]);
    }
}

{
    printf("<<<<   Array 1: Basic Operations on one-dimensional array.   >>>>\n");
    int size,a[100];
    printf("Enter the size of the Array:\n");
    scanf("%d", &size);
    printf("Enter the elements of the Array:\n");
    for(int i=0; i<size;i++)
    {
        scanf("%d",&a[i]);
    }
    int ch=1;             //ch is for choice in the menu.
    while(ch!=3)
    {
        printf("\n-------------------------------------------------\n");
        printf("Enter your choice:\n"); //menu of operations to be done on the array.
        printf("~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~\n");
        printf(" 1 Linear Search\n 2 Binary Search\n 3 Exit\n");
        printf("-------------------------------------------------\n");
        scanf("%d",&ch);
        if(ch==1)
        {
            display(a,&size);
            linearSearch(a,&size);
        }
        else if(ch==2)
        {
            display(a,&size);
