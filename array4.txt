Quesion: program in C to copy the elements of one array into another array.
Question link: https://www.w3resource.com/c-programming-exercises/array/index.php
Solution: 
#include <stdio.h>
#include <stdlib.h>
//program in C to copy the elements of one array into another array.
int main()
{
    int a[50],n;
    printf("Enter the size of the array One:\n");
    scanf("%d", &n);
    printf("Enter the elements of the Array One:\n");
    for(int i=0;i<n;i++)
    {
        scanf("%d",&a[i]);
    }
    printf("\nThe elements of the Array One:\n");
    for(int i=0;i<n;i++)
    {
        printf("%d  ",a[i]);
    }
    int b[50];
    for(int i=0;i<n;i++)
    {
        b[i]=a[i];
    }
    printf("\nThe elements of the Array Two:\n");
    for(int i=0;i<n;i++)
    {
        printf("%d  ",b[i]);
    }
    return 0;
}
