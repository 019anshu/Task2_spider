Question: Write a program in C to read n number of values in an array and display it in reverse order.
Question link: https://www.w3resource.com/c-programming-exercises/array/index.php
Solution:
#include <stdio.h>
#include <stdlib.h>
//program in C to read n number of values in an array and display it in reverse order
int main()
{
    int a[50],n;
    printf("Enter the size of the array:\n");
    scanf("%d", &n);
    printf("Enter the elements of the Array:\n");
    for(int i=0;i<n;i++)
    {
        scanf("%d",&a[i]);
    }
    printf("Printing the array in reverse order:\n");
    for(int i=n-1;i>=0;i--)
    {
        printf("%d  ",a[i]);
    }
    return 0;
}
